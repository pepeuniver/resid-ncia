function getWeek(off){

  const today=new Date(),dow=today.getDay(),mon=new Date(today);

  mon.setDate(today.getDate()-(dow===0?6:dow-1)+(off*7));

  return Array.from({length:7},(_,i)=>{const d=new Date(mon);d.setDate(mon.getDate()+i);return d});

}

function getSubjectForDay(dayIndex, slotIndex, weekOffset){

  // Use a deterministic rotation based on priority order

  const base=(weekOffset*14)+(dayIndex*2)+slotIndex;

  const notStudied=PRIORITY_SUBJECTS.filter(id=>{

    const s=gs(id);return s.status==='nao-estudado'||s.status==='estudando';

  });

  const pool=notStudied.length?notStudied:PRIORITY_SUBJECTS;

  const idx=base%pool.length;

  return ALL.find(s=>s.id===pool[idx]);

}

function getQSubject(dayIndex,weekOffset){

  // Questions on already-studied subjects

  const studied=PRIORITY_SUBJECTS.filter(id=>['estudei','questoes'].includes(gs(id).status));

  if(!studied.length){const p=PRIORITY_SUBJECTS[(dayIndex+(weekOffset*7))%PRIORITY_SUBJECTS.length];return ALL.find(s=>s.id===p)}

  const idx=((dayIndex*3)+(weekOffset*5))%studied.length;

  return ALL.find(s=>s.id===studied[idx]);

}

function getDueSR(date){

  return ALL.filter(s=>{

    const nd=getNextSRDate(s.id);

    return nd&&dk(nd)===dk(date);

  }).slice(0,2);

}

  pts.forEach(p=>{const col=p.pct>=70?'#4ade80':p.pct>=50?'#f59e0b':'#f87171';lCtx.fillStyle=col;lCtx.beginPath();lCtx.arc(p.x,p.y,4,0,Math.PI*2);lCtx.fill()});

}
